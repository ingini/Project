import json
import boto3

def lambda_handler(event, context):

    instances = ['i-01e708d23d8322370']
    ec2 = boto3.client('ec2', region_name='ap-northeast-2')
    
    ec2.start_instances(InstanceIds=instances)
    return {
        'statusCode': 200,
        'body': json.dumps('successfully started')
    }
